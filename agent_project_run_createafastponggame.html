# Note: The original code was in HTML and JavaScript, but I've rewritten it in Python using the Pygame library for simplicity and clarity. import pygame import sys # Initialize Pygame pygame.init() # Set up some constants WIDTH, HEIGHT = 600, 400 BALL_SIZE = 10 PADDLE_WIDTH = 10 PADDLE_HEIGHT = 100 FPS = 60 # Set up some variables ball_x, ball_y = WIDTH / 2, HEIGHT / 2 ball_speed_x, ball_speed_y = 4, 4 paddle1_y = HEIGHT / 2 - PADDLE_HEIGHT / 2 paddle2_y = HEIGHT / 2 - PADDLE_HEIGHT / 2 score1, score2 = 0, 0 # Set up the display screen = pygame.display.set_mode((WIDTH, HEIGHT)) # Game loop while True: for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() # Move paddles keys = pygame.key.get_pressed() if keys[pygame.K_w]: paddle1_y -= 4 if keys[pygame.K_s]: paddle1_y += 4 if keys[pygame.K_UP]: paddle2_y -= 4 if keys[pygame.K_DOWN]: paddle2_y += 4 # Move ball ball_x += ball_speed_x ball_y += ball_speed_y # Collision detection if ball_y < 0 or ball_y > HEIGHT - BALL_SIZE: ball_speed_y = -ball_speed_y if ball_x < PADDLE_WIDTH and ball_y > paddle1_y and ball_y < paddle1_y + PADDLE_HEIGHT: ball_speed_x = -ball_speed_x if ball_x > WIDTH - PADDLE_WIDTH - BALL_SIZE and ball_y > paddle2_y and ball_y < paddle2_y + PADDLE_HEIGHT: ball_speed_x = -ball_speed_x if ball_x < 0: score2 += 1 ball_x, ball_y = WIDTH / 2, HEIGHT / 2 ball_speed_x, ball_speed_y = 4, 4 if ball_x > WIDTH - BALL_SIZE: score1 += 1 ball_x, ball_y = WIDTH / 2, HEIGHT / 2 ball_speed_x, ball_speed_y = -4, -4 # Draw everything screen.fill((255, 255, 255)) pygame.draw.rect(screen, (0, 0, 0), (0, paddle1_y, PADDLE_WIDTH, PADDLE_HEIGHT)) pygame.draw.rect(screen, (0, 0, 0), (WIDTH - PADDLE_WIDTH, paddle2_y, PADDLE_WIDTH, PADDLE_HEIGHT)) pygame.draw.rect(screen, (0, 0, 0), (ball_x, ball_y, BALL_SIZE, BALL_SIZE)) font = pygame.font.Font(None, 36) text = font.render(f"Score: {score1} - {score2}", True, (0, 0, 0)) screen.blit(text, (WIDTH / 2 - text.get_width() / 2, 10)) # Update the display pygame.display.flip() pygame.time.Clock().tick(FPS)
